import math import time import sys # ANSI коды для цветов (чтобы работало везде без установки библиотек) class Colors: PURPLE = '\033[95m' CYAN = '\033[96m' DARKCYAN = '\033[36m' BLUE = '\033[94m' GREEN = '\033[92m' YELLOW = '\033[93m' RED = '\033[91m' BOLD = '\033[1m' UNDERLINE = '\033[4m' END = '\033[0m' def print_slow(text: str, delay: float = 0.03) -> None: """Красивый эффект печатной машинки""" for char in text: sys.stdout.write(char) sys.stdout.flush() time.sleep(delay) print() def draw_box(title: str, content: str) -> None: """Рисует красивую рамку вокруг текста""" width = 60 padding = 2 # Верхняя граница print(f"\n{Colors.CYAN}╔{'═' * (width - 2)}╗{Colors.END}") # Заголовок title_line = f" {Colors.BOLD}{title}{Colors.END}" # Центрирование заголовка (примерное) print(f"{Colors.CYAN}║{Colors.END} {title_line:^{width - 4}} {Colors.CYAN}║{Colors.END}") # Разделитель print(f"{Colors.CYAN}╟{'─' * (width - 2)}╢{Colors.END}") # Контент for line in content.split('\n'): print(f"{Colors.CYAN}║{Colors.END} {line:<{width - 6}} {Colors.CYAN}║{Colors.END}") # Нижняя граница print(f"{Colors.CYAN}╚{'═' * (width - 2)}╝{Colors.END}\n") def main() -> None: # Очищаем консоль (опционально, работает не везде одинаково) # os.system('cls' if os.name == 'nt' else 'clear') title = "✨ THE BEAUTY OF NUMBERS ✨" # Вычисляем число. 0.69 это примерно ln(2) magic_number = math.log(2) formatted_number = f"{magic_number:.2f}" content = f""" Вычисляем фундаментальную константу... Натуральный логарифм от 2: {Colors.YELLOW}{Colors.BOLD}ln(2) ≈ {formatted_number}{Colors.END} Статус: {Colors.GREEN}ПО КРАСОТЕ ✅{Colors.END} """ print_slow(f"{Colors.DARKCYAN}Загрузка модулей эстетики...{Colors.END}", 0.05) time.sleep(0.5) draw_box(title, content) # Финальный аккорд print(f"{Colors.PURPLE}>>> {Colors.BOLD}0.69{Colors.END} {Colors.PURPLE}достигнуто.{Colors.END}") if __name__ == "__main__": try: main() except KeyboardInterrupt: print(f"\n{Colors.RED}Прервано пользователем.{Colors.END}") Нажмите, чтобы скачать или открыть презентацию: Презентация в формате .pptx Нажмите, чтобы скачать или открыть index.htrl: Файл проверки работы с кириллицей .html
