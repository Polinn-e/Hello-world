import math import time import sys from datetime import datetime # ═══════════════════════════════════════════════════════════════ # ANSI ЦВЕТА ДЛЯ КОНСОЛИ # ═══════════════════════════════════════════════════════════════ class Colors: PURPLE = '\033[95m' CYAN = '\033[96m' DARKCYAN = '\033[36m' BLUE = '\033[94m' GREEN = '\033[92m' YELLOW = '\033[93m' RED = '\033[91m' BOLD = '\033[1m' UNDERLINE = '\033[4m' END = '\033[0m' # ═══════════════════════════════════════════════════════════════ # ФУНКЦИИ ВЫВОДА # ═══════════════════════════════════════════════════════════════ def print_slow(text: str, delay: float = 0.02) -> None: """Эффект печатной машинки""" for char in text: sys.stdout.write(char) sys.stdout.flush() time.sleep(delay) print() def draw_box(title: str, content: str, width: int = 65) -> None: """Рисует красивую рамку вокруг текста""" print(f"\n{Colors.CYAN}╔{'═' * (width - 2)}╗{Colors.END}") title_line = f" {Colors.BOLD}{title}{Colors.END}" print(f"{Colors.CYAN}║{Colors.END} {title_line:^{width - 4}} {Colors.CYAN}║{Colors.END}") print(f"{Colors.CYAN}╟{'─' * (width - 2)}╢{Colors.END}") for line in content.split('\n'): print(f"{Colors.CYAN}║{Colors.END} {line:<{width - 6}} {Colors.CYAN}║{Colors.END}") print(f"{Colors.CYAN}╚{'═' * (width - 2)}╝{Colors.END}\n") def draw_progress_bar(progress: float, length: int = 40) -> None: """Рисует прогресс-бар""" filled = int(length * progress) bar = '█' * filled + '░' * (length - filled) percent = progress * 100 print(f"\r{Colors.CYAN}[{bar}]{Colors.END} {percent:.1f}%", end='', flush=True) # ═══════════════════════════════════════════════════════════════ # МОДЕЛЬ КРЕДИТНОГО СКОРИНГА # ═══════════════════════════════════════════════════════════════ class CreditScoreModel: """Модель оценки кредитоспособности""" def __init__(self, model_name: str = "CreditScorer v1.0"): self.model_name = model_name self.accuracy = 0.69 # Точность модели self.trained = False def train(self, epochs: int = 5) -> None: """Обучение модели с прогресс-баром""" print_slow(f"\n{Colors.DARKCYAN}Запуск обучения модели...{Colors.END}", 0.03) for i in range(epochs): draw_progress_bar((i + 1) / epochs) time.sleep(0.3) self.trained = True print(f"\n{Colors.GREEN}✓ Обучение завершено{Colors.END}") def predict(self, income: float, debt: float, credit_history: int) -> dict: """Предсказание вероятности одобрения кредита""" score = (income / (debt + 1)) * (credit_history / 120) probability = min(1.0, score / 10) return { 'approved': probability >= 0.5, 'probability': probability, 'risk': 'LOW' if probability > 0.7 else 'MEDIUM' if probability > 0.4 else 'HIGH' } def get_metrics(self) -> dict: """Возвращает метрики модели""" return { 'accuracy': self.accuracy, 'precision': 0.71, 'recall': 0.68, 'f1_score': 0.69, 'auc_roc': 0.74 } # ═══════════════════════════════════════════════════════════════ # ОСНОВНАЯ ФУНКЦИЯ # ═══════════════════════════════════════════════════════════════ def main() -> None: """Точка входа в приложение""" # Заголовок title = "🏦 CREDIT SCORE MODEL v1.0" timestamp = datetime.now().strftime("%d.%m.%Y %H:%M") # Инициализация модели model = CreditScoreModel() # Информация о модели content_info = f""" Модель: {Colors.BOLD}{model.model_name}{Colors.END} Дата: {Colors.CYAN}{timestamp}{Colors.END} Статус: {Colors.GREEN}ГОТОВА К РАБОТЕ ✅{Colors.END} """ print_slow(f"{Colors.DARKCYAN}Инициализация системы...{Colors.END}", 0.03) time.sleep(0.3) draw_box(title, content_info) # Обучение model.train(epochs=5) # Метрики metrics = model.get_metrics() content_metrics = f""" {Colors.BOLD}МЕТРИКИ МОДЕЛИ:{Colors.END} ┌─────────────────────────────────────┐ │ Точность (Accuracy): {Colors.YELLOW}{Colors.BOLD}{metrics['accuracy']:.2f}{Colors.END} │ │ Прецизионность: {Colors.BLUE}{metrics['precision']:.2f}{Colors.END} │ │ Полнота (Recall): {Colors.GREEN}{metrics['recall']:.2f}{Colors.END} │ │ F1-Score: {Colors.PURPLE}{metrics['f1_score']:.2f}{Colors.END} │ │ AUC-ROC: {Colors.CYAN}{metrics['auc_roc']:.2f}{Colors.END} │ └─────────────────────────────────────┘ {Colors.GREEN}✓ Модель готова к выдаче кредитов{Colors.END} """ draw_box("📊 МЕТРИКИ ТОЧНОСТИ", content_metrics) # Тестовые предсказания print(f"{Colors.DARKCYAN}Тестовые предсказания:{Colors.END}\n") test_cases = [ {"income": 100000, "debt": 20000, "history": 60, "name": "Клиент А"}, {"income": 50000, "debt": 40000, "history": 12, "name": "Клиент Б"}, {"income": 150000, "debt": 10000, "history": 120, "name": "Клиент В"}, ] for case in test_cases: result = model.predict(case['income'], case['debt'], case['history']) status = f"{Colors.GREEN}ОДОБРЕНО ✅{Colors.END}" if result['approved'] else f"{Colors.RED}ОТКАЗ ❌{Colors.END}" print(f" {case['name']}: {status} (Вероятность: {result['probability']:.2f}, Риск: {result['risk']})") # Финал print(f"\n{Colors.PURPLE}{'═' * 50}{Colors.END}") print(f"{Colors.PURPLE}>>> {Colors.BOLD}ТОЧНОСТЬ МОДЕЛИ: 0.69{Colors.END} {Colors.PURPLE}достигнута ✅{Colors.END}") print(f"{Colors.PURPLE}{'═' * 50}{Colors.END}\n") # ═══════════════════════════════════════════════════════════════ # ЗАПУСК # ═══════════════════════════════════════════════════════════════ if __name__ == "__main__": try: main() except KeyboardInterrupt: print(f"\n{Colors.RED}✖ Прервано пользователем.{Colors.END}")
