import sys import time # ANSI коды для цветов (единый стиль) class Colors: PURPLE = '\033[95m' CYAN = '\033[96m' DARKCYAN = '\033[36m' BLUE = '\033[94m' GREEN = '\033[92m' YELLOW = '\033[93m' RED = '\033[91m' BOLD = '\033[1m' END = '\033[0m' def print_slow(text: str, delay: float = 0.05) -> None: """Эффект печатной машинки""" for char in text: sys.stdout.write(char) sys.stdout.flush() time.sleep(delay) print() def draw_centered_box(text: str) -> None: """Рисует рамку вокруг текста""" width = 50 padding = 2 print(f"\n{Colors.CYAN}╔{'═' * (width - 2)}╗{Colors.END}") print(f"{Colors.CYAN}║{Colors.END}{' ' * (width - 2)}{Colors.CYAN}║{Colors.END}") # Центрирование текста padded_text = f" {Colors.BOLD}{Colors.YELLOW}{text}{Colors.END} " print(f"{Colors.CYAN}║{Colors.END}{padded_text:^{width - 2}}{Colors.CYAN}║{Colors.END}") print(f"{Colors.CYAN}║{Colors.END}{' ' * (width - 2)}{Colors.CYAN}║{Colors.END}") print(f"{Colors.CYAN}╚{'═' * (width - 2)}╝{Colors.END}\n") def main() -> None: # Эффект загрузки print_slow(f"{Colors.DARKCYAN}Инициализация системы...{Colors.END}", 0.03) time.sleep(0.3) # Основное сообщение draw_centered_box("ПРИВЕТ, МИР! 👋") # Дополнительное сообщение print(f"{Colors.GREEN}✓{Colors.END} Статус: {Colors.BOLD}Успешно{Colors.END}") print(f"{Colors.BLUE}ℹ{Colors.END} Версия: {Colors.BOLD}0.69{Colors.END}") print(f"{Colors.PURPLE}♥{Colors.END} Код: {Colors.BOLD}По Красоте{Colors.END}\n") if __name__ == "__main__": try: main() except KeyboardInterrupt: print(f"\n{Colors.RED}✖ Завершено.{Colors.END}")
