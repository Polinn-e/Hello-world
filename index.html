import math import time import sys # ANSI коды для цветов class Colors: PURPLE = '\033[95m' CYAN = '\033[96m' DARKCYAN = '\033[36m' BLUE = '\033[94m' GREEN = '\033[92m' YELLOW = '\033[93m' RED = '\033[91m' BOLD = '\033[1m' UNDERLINE = '\033[4m' END = '\033[0m' def print_slow(text: str, delay: float = 0.03) -> None: """Красивый эффект печатной машинки""" for char in text: sys.stdout.write(char) sys.stdout.flush() time.sleep(delay) print() def draw_box(title: str, content: str) -> None: """Рисует красивую рамку вокруг текста""" width = 60 print(f"\n{Colors.CYAN}╔{'═' * (width - 2)}╗{Colors.END}") title_line = f" {Colors.BOLD}{title}{Colors.END}" print(f"{Colors.CYAN}║{Colors.END} {title_line:^{width - 4}} {Colors.CYAN}║{Colors.END}") print(f"{Colors.CYAN}╟{'─' * (width - 2)}╢{Colors.END}") for line in content.split('\n'): print(f"{Colors.CYAN}║{Colors.END} {line:<{width - 6}} {Colors.CYAN}║{Colors.END}") print(f"{Colors.CYAN}╚{'═' * (width - 2)}╝{Colors.END}\n") def main() -> None: title = "✨ THE BEAUTY OF NUMBERS ✨" magic_number = math.log(2) formatted_number = f"{magic_number:.2f}" content = f""" Вычисляем фундаментальную константу... Натуральный логарифм от 2: {Colors.YELLOW}{Colors.BOLD}ln(2) ≈ {formatted_number}{Colors.END} Статус: {Colors.GREEN}ПО КРАСОТЕ ✅{Colors.END} """ print_slow(f"{Colors.DARKCYAN}Загрузка модулей эстетики...{Colors.END}", 0.05) time.sleep(0.5) draw_box(title, content) print(f"{Colors.PURPLE}>>> {Colors.BOLD}0.69{Colors.END} {Colors.PURPLE}достигнуто.{Colors.END}") if __name__ == "__main__": try: main() except KeyboardInterrupt: print(f"\n{Colors.RED}Прервано пользователем.{Colors.END}")
